# Migration Agent - Playwright-BDD (bddgen)

## IMPORTANT: We use playwright-bdd with bddgen

- **NOT** playwright-cucumber
- Uses `npx bddgen` to generate tests from features
- Uses `createBdd()` for Given/When/Then
- Uses `defineBddConfig()` in playwright.config.ts

---

## ON "start migration"

1. Run: `find ./_source-java -name "*.java" -type f`
2. Start converting files immediately
3. Generate fixtures.ts after all pages
4. Generate playwright.config.ts with bddgen config

## ON "resume"

1. Read migration-state.json
2. Continue from next incomplete file

## 1. PAGE OBJECT TEMPLATE

```typescript
/**
 * @fileoverview Account List Page - Displays and manages user accounts
 * @author Migration Toolkit
 * @version 1.0.0
 * 
 * Migrated from: AccountListPage.java
 */

import { Page, Locator, expect } from '@playwright/test';

/**
 * Page object for the Account List page.
 * 
 * @example
 * ```typescript
 * const accountPage = new AccountListPage(page);
 * await accountPage.searchAccount('John');
 * ```
 */
export class AccountListPage {
  /** Main page instance */
  readonly page: Page;

  /** Search input field for filtering accounts */
  readonly searchInput: Locator;

  /** Submit button to apply search */
  readonly searchButton: Locator;

  /** Table containing account list */
  readonly accountsTable: Locator;

  /**
   * Creates an instance of AccountListPage.
   * @param {Page} page - Playwright page instance
   */
  constructor(page: Page) {
    this.page = page;
    this.searchInput = page.getByPlaceholder('Search accounts');
    this.searchButton = page.getByRole('button', { name: 'Search' });
    this.accountsTable = page.locator('#accounts-table');
  }

  /**
   * Navigates to the Account List page.
   * @returns {Promise<void>}
   */
  async navigate(): Promise<void> {
    await this.page.goto('/accounts');
  }

  /**
   * Searches for accounts by the given query.
   * @param {string} query - Search term to filter accounts
   * @returns {Promise<void>}
   */
  async searchAccount(query: string): Promise<void> {
    await this.searchInput.fill(query);
    await this.searchButton.click();
  }

  /**
   * Retrieves the count of accounts displayed.
   * @returns {Promise<number>} Number of account rows
   */
  async getAccountCount(): Promise<number> {
    return await this.accountsTable.locator('tbody tr').count();
  }
}
```

// Import all page objects
import { LoginPage } from '../pages/login.page';

import { TradePage } from '../pages/trade.page';

## 3. STEP DEFINITIONS (bddgen style)

```typescript
/**
 * @fileoverview Account Step Definitions - BDD steps for account functionality
 * @author Migration Toolkit
 * @version 1.0.0
 * 
 * Migrated from: AccountSteps.java
 */

import { Given, When, Then, expect } from './fixtures';

/**
 * Step: User navigates to accounts page
 */
Given('user is on the accounts page', async ({ accountListPage }) => {
  await accountListPage.navigate();
});

/**
 * Step: User is logged in and on accounts page
 */
Given('user is logged in and viewing accounts', async ({ loginPage, accountListPage }) => {
  await loginPage.navigate();
  await loginPage.login('testuser', 'password');
  await accountListPage.navigate();
});

/**
 * Step: User searches for an account
 */
When('user searches for account {string}', async ({ accountListPage }, searchTerm: string) => {
  await accountListPage.searchAccount(searchTerm);
});

/**
 * Step: User clicks on specific account
 */
When('user clicks on account {string}', async ({ accountListPage }, accountName: string) => {
  await accountListPage.clickAccountByName(accountName);
});

/**
 * Step: User filters by account type
 */
When('user filters by account type {string}', async ({ accountListPage }, accountType: string) => {
  await accountListPage.filterByType(accountType);
});

/**
 * Step: Verify account count
 */
Then('user should see {int} account(s)', async ({ accountListPage }, expectedCount: number) => {
  const actualCount = await accountListPage.getAccountCount();
  expect(actualCount).toBe(expectedCount);
});

/**
 * Step: Verify account list is displayed
 */
Then('the account list should be displayed', async ({ accountListPage }) => {
  await expect(accountListPage.accountsTable).toBeVisible();
});

/**
 * Step: Verify specific account is visible
 */
Then('user should see account {string} in the list', async ({ accountListPage }, accountName: string) => {
  await expect(accountListPage.accountsTable.getByText(accountName)).toBeVisible();
});
```

## 4. PLAYWRIGHT.CONFIG.TS (with defineBddConfig)

```typescript
/**
 * @fileoverview Playwright Configuration with BDD (bddgen) support
 * @author Migration Toolkit
 * @version 1.0.0
 */

import { defineConfig, devices } from '@playwright/test';
import { defineBddConfig, cucumberReporter } from 'playwright-bdd';

/**
 * BDD Configuration
 * - features: Path to .feature files
 * - steps: Path to step definition files
 */
const testDir = defineBddConfig({
  features: 'features/**/*.feature',
  steps: 'src/steps/**/*.ts',
});

export default defineConfig({
  testDir,
  
  /* Maximum time one test can run */
  timeout: 30 * 1000,
  
  /* Retry on CI only */
  retries: process.env.CI ? 2 : 0,
  
  /* Reporter configuration */
  reporter: [
    ['html', { open: 'never' }],
    ['list'],
    cucumberReporter('html', { outputFile: 'cucumber-report/report.html' }),
  ],
  
  /* Shared settings for all projects */
  use: {
    /* Base URL for navigation */
    baseURL: process.env.BASE_URL || 'http://localhost:3000',
    
    /* Capture screenshot on failure */
    screenshot: 'only-on-failure',
    
    /* Record trace on failure */
    trace: 'retain-on-failure',
    
    /* Viewport size */
    viewport: { width: 1280, height: 720 },
  },

  /* Configure projects for major browsers */
  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
  ],
});

## CONVERSION RULES

### Locators
```
By.xpath("//button[text()='X']")     → getByRole('button', { name: 'X' })
By.xpath("//a[text()='X']")          → getByRole('link', { name: 'X' })
By.xpath("//input[@id='X']")         → locator('#X')
By.xpath("//*[@data-testid='X']")    → getByTestId('X')
By.xpath("//input[@placeholder='X']") → getByPlaceholder('X')
```

### Step Annotations → playwright-bdd
```java
// Java (Cucumber)
@Given("user is on login page")
public void userOnLoginPage() {
    loginPage.navigate();
}
```

```typescript
// TypeScript (playwright-bdd)
Given('user is on login page', async ({ loginPage }) => {
  await loginPage.navigate();
});
```

### @And/@But Conversion
```
@And with click/enter/select  → When
@And with should/verify/see   → Then
@And with is/has/already      → Given
@But                          → Then
```

## MIGRATION ORDER

1. **Pages** → `src/pages/*.page.ts`
2. **Fixtures** → `src/steps/fixtures.ts`
3. **Steps** → `src/steps/*.steps.ts`
4. **Features** → `features/*.feature`
5. **Config** → `playwright.config.ts`
6. **Update** → `package.json` scripts
